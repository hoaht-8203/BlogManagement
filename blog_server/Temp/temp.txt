-- Bảng Departments (Các đơn vị/bộ môn)
CREATE TABLE Departments (
    Id INT PRIMARY KEY,
    Name NVARCHAR(255),
    Description NTEXT,
    ParentDepartmentId INT,
    Type NVARCHAR(50) -- (Bộ môn, Văn phòng, Phòng thí nghiệm, etc.)
);

-- Bảng Programs (Chương trình đào tạo)
CREATE TABLE Programs (
    Id INT PRIMARY KEY,
    Name NVARCHAR(255),
    Code VARCHAR(50), -- Ví dụ: CTTA07
    Description NTEXT,
    AdmissionQuota INT, -- Chỉ tiêu tuyển sinh
    Type NVARCHAR(50) -- (Đại học, Sau đại học, Ngắn hạn)
);

-- Bảng Subjects (Môn học)
CREATE TABLE Subjects (
    Id INT PRIMARY KEY,
    Name NVARCHAR(255),
    Code VARCHAR(50),
    Credits INT,
    ProgramId INT
);

-- Bảng News (Tin tức - thông báo)
CREATE TABLE News (
    Id INT PRIMARY KEY,
    Title NVARCHAR(255),
    Content NTEXT,
    PublishDate DATETIME,
    Type NVARCHAR(50), -- (Tin tức, Thông báo, Tuyển dụng)
    ViewCount INT
);

-- Bảng Research (Nghiên cứu khoa học)
CREATE TABLE Research (
    Id INT PRIMARY KEY,
    Title NVARCHAR(255),
    Description NTEXT,
    Type NVARCHAR(50), -- (Đề tài NCKH, Bài báo, Hội thảo)
    Year INT,
    Authors NVARCHAR(MAX)
);

-- Bảng Events (Sự kiện)
CREATE TABLE Events (
    Id INT PRIMARY KEY,
    Title NVARCHAR(255),
    Description NTEXT,
    StartTime DATETIME,
    EndTime DATETIME,
    Location NVARCHAR(255),
    Type NVARCHAR(50) -- (Hội thảo, Lễ trao bằng, Hoạt động ngoại khóa)
);

-- Bảng Partners (Đối tác)
CREATE TABLE Partners (
    Id INT PRIMARY KEY,
    Name NVARCHAR(255),
    Description NTEXT,
    Website NVARCHAR(255),
    Type NVARCHAR(50) -- (Doanh nghiệp, Trường đại học, etc.)
);

-- Bảng Faculty (Thông tin về Viện/Khoa)
CREATE TABLE Faculty (
    Id INT PRIMARY KEY,
    Name NVARCHAR(255),
    Description NTEXT,
    Address NVARCHAR(500),
    Phone NVARCHAR(50),
    Email NVARCHAR(100),
    WorkingHours NVARCHAR(255)
);

-- Bảng Departments (Bộ môn)
CREATE TABLE Departments (
    Id INT PRIMARY KEY,
    Name NVARCHAR(255),
    Description NTEXT,
    HeadOfDepartment NVARCHAR(255)
);

-- Bảng EducationPrograms (Chương trình đào tạo)
CREATE TABLE EducationPrograms (
    Id INT PRIMARY KEY,
    Name NVARCHAR(255),
    Code VARCHAR(20), -- Ví dụ: EP04, EP12
    Level NVARCHAR(50), -- Đại học, Sau đại học, Ngắn hạn
    Type NVARCHAR(50), -- Chương trình thường, Quốc tế
    Description NTEXT,
    CertificationIntegration NVARCHAR(255) -- Ví dụ: ICAEW CFAB
);

-- Bảng News (Tin tức & Sự kiện)
CREATE TABLE News (
    Id INT PRIMARY KEY,
    Title NVARCHAR(255),
    Content NTEXT,
    PublishDate DATETIME,
    Category NVARCHAR(50), -- Tin tức, Thông báo, Sự kiện
    ImageUrl NVARCHAR(500),
    ViewCount INT
);

-- Bảng Scholarships (Học bổng)
CREATE TABLE Scholarships (
    Id INT PRIMARY KEY,
    Name NVARCHAR(255),
    Description NTEXT,
    Deadline DATETIME,
    Value DECIMAL(18,2),
    Sponsor NVARCHAR(255),
    Requirements NTEXT
);

-- Bảng JobOpportunities (Cơ hội việc làm)
CREATE TABLE JobOpportunities (
    Id INT PRIMARY KEY,
    CompanyName NVARCHAR(255),
    Position NVARCHAR(255),
    Description NTEXT,
    Requirements NTEXT,
    PostDate DATETIME,
    Deadline DATETIME,
    Type NVARCHAR(50) -- Fulltime, Internship, etc.
);

-- Bảng Staff (Cán bộ, giảng viên)
CREATE TABLE Staff (
    Id INT PRIMARY KEY,
    Name NVARCHAR(255),
    Title NVARCHAR(100),
    DepartmentId INT,
    Email NVARCHAR(100),
    Phone NVARCHAR(50),
    AcademicDegree NVARCHAR(50),
    Position NVARCHAR(100)
);

-- Bảng ResearchActivities (Hoạt động nghiên cứu khoa học)
CREATE TABLE ResearchActivities (
    Id INT PRIMARY KEY,
    Title NVARCHAR(255),
    Type NVARCHAR(50), -- Hội thảo, Tọa đàm, NCKH Sinh viên
    Description NTEXT,
    Date DATETIME,
    Organizer NVARCHAR(255)
);

-- Bảng MediaLibrary (Thư viện media)
CREATE TABLE MediaLibrary (
    Id INT PRIMARY KEY,
    Title NVARCHAR(255),
    Type NVARCHAR(50), -- Video, Image, Document
    Url NVARCHAR(500),
    Description NTEXT,
    UploadDate DATETIME
);
